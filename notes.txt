// Added session support for login functionality
const session = require('express-session');

// Session middleware configuration for user authentication
app.use(session({
  secret: 'dog-walking-secret-key',
  resave: false,
  saveUninitialized: false,
  cookie: { secure: false, maxAge: 24 * 60 * 60 * 1000 } // 24 hours session duration
}));

// Authentication middleware to protect routes that require login
function requireAuth(req, res, next) {
  if (req.session.user) {
    next(); // User is authenticated, proceed to route
  } else {
    res.status(401).json({ error: 'Authentication required' });
  }
}
